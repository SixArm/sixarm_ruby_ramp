def self.cartesian_product(*enums)
  result = [[]]
  while [] != enums
    t, result = result, []
    b, *enums = enums
    t.each do |a|
      b.each do |n|
        result << a + [n]
      end
    end
  end
  result
end

def bisect
  accept=[]
  reject=[]
  each{|item|
   if yield(item)
    accept << item
   else
    reject << item
   end
  }
  return accept,reject
end

def mutex?
  num = 0
  each{|item|
    if yield(item)
      num += 1
      if num > 1 then return false end
    end
  }
  return true
end

def to_h_merge
  hash={}
  seen=Set.new
  each{|key,val|
    if hash.key? key
      if seen.include? key
        hash[key] << val
      else
        hash[key]=[hash[key]]
        hash[key] << val
        seen << key
      end
    else
      hash[key]=val
    end
  }
  return hash
end
