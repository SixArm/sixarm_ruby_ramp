def self.cartesian_product(*enums)
  result = [[]]
  while [] != enums
    t, result = result, []
    b, *enums = enums
    t.each do |a|
      b.each do |n|
        result << a + [n]
      end
    end
  end
  result
end

def bisect
  accept=[]
  reject=[]
  each{|item| 
   if yield(item)
    accept << item
   else
    reject << item
   end
  }
  return accept,reject
end
