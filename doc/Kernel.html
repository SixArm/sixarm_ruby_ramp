def pp_s(*objs)  
  str = StringIO.new  
  objs.each {|obj|  
      PP.pp(obj, str)  
    }  
    str.rewind  
    str.read  
end

def caller_method_name(caller_index=0)
  RUBY_VERSION > '2.0' \
  ? caller_locations(caller_index + 1,1).first.base_label \
  : caller[caller_index][/`([^']*)'/, 1]
end
