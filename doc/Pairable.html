Overview

Extensions for a class that provides key-value pairs.

The including class must provide these methods:

* obj[key] → value
* obj[key] = value
* keys → enumerable

def each_key!
  replacements=[]
  keys.each{|key|
    value=self[key]
    key2=yield(key)
    if key===key2
      #nop
    else
      replacements << [key,key2,value]
    end
  }
  replacements.each{|key,key2,value|
    self.delete(key)
    self[key2]=value
  }
  return self
end

def each_pair!
  replacements=[]
  keys.each{|key|
    value=self[key]
    key2,value2=yield(key,value)
    if key===key2
      if value===value2
        #nop
      else
        self[key]=value2
      end
    else
      replacements << [key,key2,value2]
    end
  }
  replacements.each{|key,key2,value2|
    self.delete(key)
    self[key2]=value2
  }
  return self
end

- (Object) each_sort

Call block once for each key-value pair in self, passing the key and value to the block as a two-element array.

The keys are sorted.

Examples:

h = { "c" => "d", "a" => "b" }
h.each_sort {|key, val| ... }
=> calls the block with "a" => "b", then with "c" => "d"

    if value===value2

    end
  }

end
